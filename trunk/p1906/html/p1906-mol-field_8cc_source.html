619  double sz2 = gsl_vector_get (segment, 2);
620 
622  gsl_vector * dV = gsl_vector_alloc (3);
623 
625  gsl_vector_set (dV, 0, sx2 - sx1);
626  gsl_vector_set (dV, 1, sy2 - sy1);
627  gsl_vector_set (dV, 2, sz2 - sz1);
628 
629  if ( ( ((px - sx1) / gsl_vector_get (dV, 0)) == ((py - sy1) / gsl_vector_get (dV, 1)) ) &&
630  ( ((py - sy1) / gsl_vector_get (dV, 1)) == ((pz - sz1) / gsl_vector_get (dV, 2)) ) )
631  overlap = true;
632
